64 def apply(self):
65 """!
66 @brief It reads, checks and prepares the entered time
67 for sending. If the data is valid, it closes
68 the window. Otherwise, it deletes the entered values.
69 @param None
70 @return None
71 """
72 # Get year. Only the last two digits are processed.
73 try:
74 year = int(self.entry_year.get())-2000
75 except:
76 return
77 else:
78 # Get month.
79 month = month2num(self.entry_month.get())
80 if (month > 12):
81 return
82
83 # Get day in the week.
84 day = self.entry_day.get()
85 if (day == ""):
86 return
87 day = day2num(day[0:2]) # Funcion processes the first two letters.
88
89 # Get date. Entry data must be integer.
90 try:
91 date = int(self.entry_date.get())
92 except:
93 # Not integer.
94 return
95 else:
96
97 # Number of days in months varies.
98 if (date < 1):
99 return
100 # No month has more than 31 days.
101 if (date > 31):
102 return
103 # February outside of leap years.
104 if (date > 28) & (month == 2) & (year%4 != 0):
105 return
106 # February in general (leap and standard years).
107 if (date > 29) & (month == 2):
108 return
109 # April, June, September and November have 30 days.
110 if (date > 30) & ((month == 4) | (month == 6) | (month == 9) | (month == 11)):
111 return
112
113 # Get hour.
114 hour = self.entry_hour.get()
115 if (hour == ""):
116 return
117 # Hour must be integer in the range of 0 to 23.
118 hour = int(hour)
119 if (hour > 23) | (hour < 0):
120 return
121
122 # Get minute. Minute must be integer.
123 try:
124 minute = int(self.entry_minute.get())
125 except:
126 # Not integer.
127 return
128 else:
129 # Minute must be in the range of 0 to 59.
130 if (minute < 0) | (minute > 59):
131 return
132
133 # Get second. Second must be integer.
134 try:
135 second = int(self.entry_second.get())
136 except:
137 # Not integer.
138 return
139 else:
140 # Second must be in the range of 0 to 59.
141 if (second < 0) | (second > 59):
142 return
143
144 # Null time packet. The first byte remains 'T'.
145 for i in range(1,8,1):
146 self.master.time_packet[i] = 0x00
147
148 # Create time packet. #
149 # The tens of the year are stored in the higher nibble and the units in the lower nibble.
150 self.master.time_packet[1] |= (((year//10)<<4) | (year%10))
151 # The tens of the month are stored in the higher nibble and the units in the lower nibble.
152 # The MSB carry information about century. 0 - 20th, 1 - 21th.
153 self.master.time_packet[2] |= (1 << 7) | (((month//10)<<4) | (month%10))
154 # The tens of the date are stored in the higher nibble and the units in the lower nibble.
155 self.master.time_packet[3] |= (((date//10)<<4) | (date%10))
156 # The day of the week corresponds to the number 0 - 6.
157 self.master.time_packet[4] |= day
158 # The tens of the hour are stored in the higher nibble and the units in the lower nibble.
159 self.master.time_packet[5] |= (((hour//10)<<4) | (hour%10))
160 # The tens of the minute are stored in the higher nibble and the units in the lower nibble.
161 self.master.time_packet[6] |= (((minute//10)<<4) | (minute%10))
162 # The tens of the second are stored in the higher nibble and the units in the lower nibble.
163 self.master.time_packet[7] |= (((second//10)<<4) | (second%10))
164
165 # Send data
166 self.master.send_time_packet()
167 self.destroy()
