29def default_data(self):
30 """!
31 @brief Creates individual data stores for the main window.
32 It fills them with default data.
33 @param None
34 @return None
35 @details The following directories are created:
36 * settings - It contains set values of controlled
37 variables and hysteresis.
38 * system_settings - It contains the allowed ranges of
39 values and hysteresis.
40 * measure_value - The received measured values of
41 the quantities will be stored here.
42
43 The following lists are created:
44 * time_packet - This is where the time setting data
45 is stored before it is sent. The first
46 byte is always the ASCII code 'T'
47 for recognition.
48 * reg_packet - This is where the regulation setting data
49 is stored before it is sent. The first
50 byte is always the ASCII code 'R'
51 for recognition.
52 It also creates a repository for connection phase information.
53 Phase: 0 - no connection.
54 Phase: 1 - preperation to start.
55 Phase: 2 - communication is running.
56 """
57 self.setting = {'temp_val':25,'temp_hys':2,'hum_val':50,'hum_hys':5,
58 'soil_val':50,'soil_hys':5,'ilu_val':100}
59 self.system_setting = {'ilu_min':20,'ilu_max':500,'temp_val_min':15,
60 'temp_val_max':50,'temp_hys_min':1,'temp_hys_max':10,
61 'hum_val_min':20,'hum_val_max':90,'hum_hys_min':5,'hum_hys_max':30,
62 'soil_val_min':5,'soil_val_max':90,'soil_hys_min':5,'soil_hys_max':30}
63 self.measure_value = {'temp':20,'hum':65.7,'soil':54,'ilu':200}
64 self.time_packet = [0x54,0x00,0x00,0x00,0x00,0x00,0x00,0x00]
65 self.reg_packet = [0x52,0x00,0x00,0x00,0x00,0x00,0x00,0x00,0x00]
