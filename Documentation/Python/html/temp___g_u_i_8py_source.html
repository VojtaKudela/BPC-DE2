72 def get_val(self):
73 """!
74 @brief Reads the set temperature value.
75 @param None
76 @return The set temperature value has been changed.
77 @retval 0 - No change, 1 - Value has be changed.
78 @details If the specified value is an int value and lies between
79 temp_val_min and temp_val_max, the value will be set
80 in the self.master.setting['temp_val'] directory.
81 """
82 try:
83 temp = int(self.entry_00.get())
84 except:
85 return 0
86 else:
87 if (temp <= self.master.system_setting['temp_val_max']) & (temp >= self.master.system_setting['temp_val_min']):
88 self.master.setting['temp_val'] = temp
89 return 1
90 else:
91 return 0

94 def get_hys(self,update):
95 """!
96 @brief Reads the set hysteresis value.
97 @param update Temperature value has changed.
98 @return None
99 @details If the specified value is an int value and lies between
100 temp_hys_min and temp_hys_max, the value will be set
101 in the self.master.setting['temp_hys'] directory. The main window
102 update is started. The allowed value is displayed in the main window.
103 The update will also take place if the hysteresis is not valid but
104 the temperature value has changed.
105 @note If communication is running, this value will be written to
106 the embedded device.
107 """
108 try:
109 temp = int(self.entry_01.get())
110 except:
111 # Invalid hysteresis but value was changed. Not number.
112 if update == 1:
113 self.get_finish()
114 else:
115 if (temp <= self.master.system_setting['temp_hys_max']) & (temp >= self.master.system_setting['temp_hys_min']):
116 # Valid hysteresis but value was changed.
117 self.master.setting['temp_hys'] = temp
118 self.get_finish()
119 else:
120 # Invalid hysteresis but value was changed. Out of range.
121 if update == 1:
122 self.get_finish()
