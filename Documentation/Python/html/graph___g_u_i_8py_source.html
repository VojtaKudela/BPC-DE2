301 def week_con(self,i):
302 """!
303 @brief It verifies whether the given day
304 was or was not during the last week.
305 @param i (int) The order of the line being read.
306 @return Was it within the last month.
307 @note It also checks the situation when
308 last week is at the turn of the month
309 or even the turn of the year.
310 """
311 # Same month
312 a1 = (int(self.data.data[-1]["month"]) == int(self.data.data[i]["month"]))
313 a2 = (int(self.data.data[-1]["year"]) == int(self.data.data[i]["year"]))
314 a3 = (int(self.data.data[-1]["date"]) - int(self.data.data[i]["date"]) < 7)
315 a4 = (int(self.data.data[-1]["date"]) - int(self.data.data[i]["date"]) >= 0)
316 a = a1 & a2 & a3 & a4
317 # Consecutive months
318 b1 = (int(self.data.data[-1]["year"]) == int(self.data.data[i]["year"]))
319 b2 = (int(self.data.data[-1]["month"]) - int(self.data.data[i]["month"]) == 1)
320 b3 = (int(self.data.data[-1]["date"]) + n_of_days(self.data.data[i]["month"],self.data.data[i]["year"]) -
321 int(self.data.data[i]["date"]) < 7)
322 b = b1 & b2 & b3
323 # Consecutive months, consecutive years
324 c1 = (int(self.data.data[-1]["year"]) - int(self.data.data[i]["year"]) == 1)
325 c2 = (int(self.data.data[-1]["month"]) == 1)
326 c3 = (int(self.data.data[i]["month"]) == 12)
327 c4 = (int(self.data.data[-1]["date"]) + 31 - int(self.data.data[i]["date"]) < 7)
328 c = c1 & c2 & c3 & c4
329 # The last condition is if all data is from last week.
330 return (a | b | c) & (abs(i) < len(self.data.data))
