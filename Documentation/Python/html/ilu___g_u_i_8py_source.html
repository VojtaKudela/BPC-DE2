53 def apply(self):
54 """!
55 @brief Sets the specified illumination value.
56 @param None
57 @return None
58 @details If the specified value is an int value and lies between
59 ilu_min and ilu_max, the value will be set
60 in the self.master.setting['ilu_val'] directory. The main window
61 update is started. The allowed value is displayed in the main window.
62 @note If communication is running, this value will be written to
63 the embedded device.
64 """
65 try:
66 temp = int(self.entry.get())
67 except:
68 return
69 else:
70 # Set value must be in allowed range.
71 if (temp <= self.master.system_setting['ilu_max']) & (temp >= self.master.system_setting['ilu_min']):
72 # Storage illumination value to main window data file.
73 self.master.setting['ilu_val'] = temp
74 # Update main window.
75 self.master.update_settings()
76 # Close illumination window.
77 self.destroy()
