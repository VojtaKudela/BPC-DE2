170 def GUI_connect(self):
171 """!
172 @brief Callback function when the Connect button
173 is pressed.
174 @param None
175 @return None
176 @details The first press starts the communication.
177 If successful, the button changes color
178 to green and the text to RUN.
179 Otherwise, it turns red.
180
181 A second press enables communication.
182 The button will change its text to Stop
183 and its color to red.
184
185 A third press stops communication.
186 The gray Connect button appears again.
187 """
188 # Communication is not started.
189 if (self.phase == 0):
190 # Try start communication.
191 flat = self.start_serial()
192 if (flat == 0):
193 # Successful start.
194 self.connect_but.configure(fg_color=self.color['run_col'],text="Run")
195 self.phase = 1 # Communication was started.
196 else:
197 # Unseccessful start.
198 self.connect_but.configure(fg_color=self.color['stop_col'],text="Connect")
199 self.phase = 0 # Communication was stopped.
200
201 # Communication is started.
202 elif (self.phase == 1):
203 self.connect_but.configure(fg_color=self.color['stop_col'],text="Stop")
204 self.phase = 2 # Now communication is running.
205
206 # Communication is running
207 elif (self.phase == 2):
208 self.stop_serial() # Close serial communication.
209 self.connect_but.configure(fg_color=self.color['gray'],text="Connect")
210 self.phase = 0 # Now communication is stopped.

287 def start_serial(self):
288 """!
289 @brief Starts serial communication.
290 @param None
291 @return None
292 """
293 try:
294 # It attempts to establish serial communication.
295 self.serial_connection = serial.Serial(self.serial_port,self.baud_rate,timeout=1)
296 # Creates a thread that will periodically check for data received by the serial port.
297 self.serial_thread = threading.Thread(target=self.read_from_serial, daemon=True)
298 self.serial_thread.start()
299 except serial.SerialException as e:
300 # Serial communication failed to start.
301 return 1
302 else:
303 # Serial communication was successfully started.
304 return 0

307 def read_from_serial(self):
308 """!
309 @brief Functions running on a separate thread.
310 Reading, displaying and saving received data.
311 @param None
312 @return None
313 """
314 # The thread ends when communication is stopped.
315 while self.running:
316 # Communication exists, it doesn't wait and it runs.
317 if self.serial_connection and self.serial_connection.in_waiting > 0 and self.phase == 2:
318 try:
319 # Attempt to read a line.
320 line = self.serial_connection.readline()
321 except Exception as e:
322 return
323 else:
324 # The received data is stored in the raw_data object.
325 raw = raw_data()
326 for i in range (0,13,1):
327 raw.add(line[i])
328 # Processing received data.
329 self.data = data_file(raw)
330 # Saving received data.
331 self.measure_value['ilu'] = self.data.lig
332 self.measure_value['soil'] = self.data.soil
333 self.measure_value['hum'] = self.data.hum
334 self.measure_value['temp'] = self.data.temp
335 # Display received data.
336 self.update_measure_value()
337 # Adding received data to the database.
338 self.data.save_data()
339 time.sleep(0.1)
