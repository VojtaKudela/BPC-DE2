71 def get_val(self):
72 """!
73 @brief Reads the set humidity value.
74 @param None
75 @return The set humidity value has been changed.
76 @retval 0 - No change, 1 - Value has be changed.
77 @details If the specified value is an int value and lies between
78 hum_val_min and hum_val_max, the value will be set
79 in the self.master.setting['hum_val'] directory.
80 """
81 try:
82 temp = int(self.entry_00.get())
83 except:
84 return 0
85 else:
86 if (temp <= self.master.system_setting['hum_val_max']) & (temp >= self.master.system_setting['hum_val_min']):
87 self.master.setting['hum_val'] = temp
88 return 1
89 else:
90 return 0

93 def get_hys(self,update):
94 """!
95 @brief Reads the set hysteresis value.
96 @param update Humidity value has changed.
97 @return None
98 @details If the specified value is an int value and lies between
99 hum_hys_min and hum_hys_max, the value will be set
100 in the self.master.setting['hum_hys'] directory. The main window
101 update is started. The allowed value is displayed in the main window.
102 The update will also take place if the hysteresis is not valid but
103 the humidity value has changed.
104 @note If communication is running, this value will be written to
105 the embedded device.
106 """
107 try:
108 temp = int(self.entry_01.get())
109 except:
110 # Invalid hysteresis but value was changed. Not number.
111 if update == 1:
112 self.get_finish()
113 else:
114 if (temp <= self.master.system_setting['hum_hys_max']) & (temp >= self.master.system_setting['hum_hys_min']):
115 # Valid hysteresis but value was changed.
116 self.master.setting['hum_hys'] = temp
117 self.get_finish()
118 else:
119 # Invalid hysteresis but value was changed. Out of range.
120 if update == 1:
121 self.get_finish()
