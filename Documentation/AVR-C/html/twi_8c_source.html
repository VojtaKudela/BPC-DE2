19/**********************************************************************
20 * Function: twi_init()
21 * Purpose: Initialize TWI unit, enable internal pull-ups, and set SCL
22 * frequency.
23 * Returns: none
24 **********************************************************************/
25void twi_init(void)
26{
27 /* Enable internal pull-up resistors */
28 DDR(TWI_PORT) &= ~((1<<TWI_SDA_PIN) | (1<<TWI_SCL_PIN));
29 TWI_PORT |= (1<<TWI_SDA_PIN) | (1<<TWI_SCL_PIN);
30
31 /* Set SCL frequency */
32 TWSR &= ~((1<<TWPS1) | (1<<TWPS0));
33 TWBR = TWI_BIT_RATE_REG;
34}

39 * Purpose: Start communication on I2C/TWI bus.

42void twi_start(void)
43{
44 /* Send Start condition */
45 TWCR = (1<<TWINT) | (1<<TWSTA) | (1<<TWEN);
46 while ((TWCR & (1<<TWINT)) == 0);

50/**********************************************************************
51 * Function: twi_write()
52 * Purpose: Send one byte to I2C/TWI Slave device.
53 * Input: data Byte to be transmitted
54 * Returns: ACK/NACK received value
55 **********************************************************************/
56uint8_t twi_write(uint8_t data)
57{
58 uint8_t twi_status;
59
60 /* Send SLA+R, SLA+W, or data byte on I2C/TWI bus */
61 TWDR = data;
62 TWCR = (1<<TWINT) | (1<<TWEN);
63 while ((TWCR & (1<<TWINT)) == 0);
64
65 /* Check value of TWI status register */
66 twi_status = TWSR & 0xf8;
67
68 /* Status Code:
69 * 0x18: SLA+W has been transmitted and ACK received
70 * 0x28: Data byte has been transmitted and ACK has been received
71 * 0x40: SLA+R has been transmitted and ACK received
72 */
73 if (twi_status == 0x18 || twi_status == 0x28 || twi_status == 0x40)
74 return 0; /* ACK received */
75 else
76 return 1; /* NACK received */
77}

80/**********************************************************************
81 * Function: twi_read()
82 * Purpose: Read one byte from I2C/TWI Slave device and acknowledge
83 * it by ACK or NACK.
84 * Input: ack ACK/NACK value to be transmitted
85 * Returns: Received data byte
86 **********************************************************************/
87uint8_t twi_read(uint8_t ack)
88{
89 if (ack == TWI_ACK)
90 TWCR = (1<<TWINT) | (1<<TWEN) | (1<<TWEA);
91 else
92 TWCR = (1<<TWINT) | (1<<TWEN);
93 while ((TWCR & (1<<TWINT)) == 0);
94
95 return (TWDR);
96}
