13/*
14 * Function: PWM_enable
15 * Purpose: Enable PWM
16 * Input(s): none
17 * Returns: none
18 */
19void PWM_enable()
20{
21 DDRD |= (1<<PD6);
22 TCCR0A |= (1 << WGM00) | (1 << WGM01) | (1 << COM0A1); // Fast PWM, non-inverted output on OC0A
23 TCCR0B &= ~(1<<CS02); TCCR0B |= (1<<CS01) | (1<<CS00); // Start timer with prescaler 64
24}

37/*
38 * Function: PWM_control
39 * Purpose: Automatically control duty
40 * Input(s): current_value - measured value
41 * set_value - required value
42 * step - step increase or decrease value
43 * Returns: none
44 */
45void PWM_control(uint16_t current_value, uint16_t set_value, uint8_t step)
46{
47 static uint8_t duty = 0;
48
49 // duty <= (255 - step) - protect overflow
50 if ((current_value < set_value) & (duty <= (255 - step)))
51 {
52 duty += step;
53 }
54 // (duty >= step) - protect underflow
55 else if ((current_value > set_value) & (duty >= step))
56 {
57 duty -= step;
58 }
59
60 PWM_set_duty(duty);
61}
