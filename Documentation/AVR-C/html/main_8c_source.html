167/*
168 * Function: read_dht_data
169 * Purpose: Reads data from the combined humidity and temperature sensor.
170 * Input(s): None
171 * Returns: None
172 */
173void read_dht_data(void)
174{
175 // Temporary storage of the read values
176 uint8_t hum_raw[2], temp_raw[2];
177 uint8_t cumsum;
178
179 twi_start();
180
181 // DTH I2C address
182 if (twi_write((DHT_ADR<<1) | TWI_WRITE) == 0) // Start communication with DHT
183 {
184 twi_write(DHT_HUM_MEM); // Point to humidity register
185 twi_stop();
186
187 // Repeated start
188 twi_start();
189 twi_write((DHT_ADR << 1) | TWI_READ); // Read operation
190 hum_raw[0] = twi_read(TWI_ACK); // Read high byte of humidity
191 hum_raw[1] = twi_read(TWI_ACK); // Read low byte of humidity
192 temp_raw[0] = twi_read(TWI_ACK); // Read high byte of temperature
193 temp_raw[1] = twi_read(TWI_ACK); // Read low byte of temperature
194 cumsum = twi_read(TWI_NACK); // Read cumsum
195
196 // Are received data correct?
197 if ((hum_raw[1] + temp_raw[1] + hum_raw[0] + temp_raw[0]) == cumsum)
198 {
199 dp.hum_int = hum_raw[0];
200 dp.hum_dec = hum_raw[1];
201 dp.temp_int = temp_raw[0];
202 dp.temp_dec = temp_raw[1];
203 }
204 }
205
206 twi_stop();
207
208}

246// Main loop
247int main(void)
248{
249 uint16_t ilu_set; // The lighting control level will be stored here.
250 uint8_t new_data; // Carry information about type of new data 0 - none, 1 - reg, 2 - time
251 uint8_t data_count; // Number of receive bytes
252 uint16_t uart_data; // Receive uart_byte
253
254 // Initialization
255 startup();
256
257 // Forever loop
258 while (1)
259 {
260 // Obtaining data from serial communication.
261 uart_data = uart_getc();
262
263 if ((uart_data>>8)==0)
264 {
265 // Only the lower byte contains data.
266 uart_data = (uart_data&0x00ff);
267
268 // First letter carry information about type of data
269 if (new_data == 0)
270 {
271
272 // The data packet with the time setting starts with the ASCII character T.
273 if (uart_data == 'T')
274 {
275 // Ready to receive time data packet.
276 new_data = 2;
277 data_count++;
278 }
279
280 // The data packet with the regulation setting starts with the ASCII character R.
281 else if (uart_data == 'R')
282 {
283
284 // Ready to receive regulation data packet.
285 new_data = 1;
286 data_count++;
287 }
288 }
289
290 // Receive regulation data.
291 // Data on the serial line has a strict order.
292 else if (new_data == 1)
293 {
294 switch (data_count)
295 {
296 case 1:
297 rg.temp_val = uart_data;
298 break;
299 case 2:
300 rg.temp_hys = uart_data;
301 break;
302 case 3:
303 rg.hum_val = uart_data;
304 break;
305 case 4:
306 rg.hum_hys = uart_data;
307 break;
308 case 5:
309 rg.soil_val = uart_data;
310 break;
311 case 6:
312 rg.soil_hys = uart_data;
313 break;
314 case 7:
315 rg.ilu_100 = uart_data;
316 break;
317 default:
318 rg.ilu_1 = uart_data;
319 break;
320 }
321
322 // Ready to process another byte.
323 data_count++;
324
325 }
326
327 // Receive time data.
328 // Data on the serial line has a strict order.
329 else if (new_data == 2)
330 {
331 switch (data_count)
332 {
333 case 1:
334 rc.year = uart_data;
335 break;
336 case 2:
337 rc.month = uart_data;
338 break;
339 case 3:
340 rc.date = uart_data;
341 break;
342 case 4:
343 rc.day = uart_data;
344 break;
345 case 5:
346 rc.hour = uart_data;
347 break;
348 case 6:
349 rc.minute = uart_data;
350 break;
351 default:
352 rc.second = uart_data;
353 break;
354 }
355
356 // Ready to process another byte.
357 data_count++;
358 }
359
360 // Apply regulation value
361 if ((new_data == 1) & (data_count > 8))
362 {
363 data_count = 0; // Ready for the next reception.
364 new_data = 0; // Nothing is received.
365
366 }
367 // Apply time value
368 else if ((new_data == 2) & (data_count > 7))
369 {
370 data_count = 0; // Ready for the next reception.
371 new_data = 0; // Nothing is received.
372 write_rtc_data(); // Data are send via I2C
373 }
374 }
375
376 // After TIMER1 overflows.
377 if (update_data == 1)
378 {
379 // Read RTC and DHT data
380 read_rtc_data();
381 read_dht_data();
382
383 // Control of PWM
384 ilu_set = rg.ilu_100*100 + rg.ilu_1;
385 PWM_control(PD_ilu(dp.lux),ilu_set,10);
386
387 // Control ventil
388 SOIL_control(REG_PORT, VALV, PD_soil(dp.soil), rg.soil_val, rg.soil_hys);
389
390 // Control heater, fan and inhalator
391 HEATER_FAN_control(REG_PORT, HEAT, REG_PORT, COOL, REG_PORT, INHA, PD_round(dp.temp_int,dp.temp_dec),
392 rg.temp_val, rg.temp_hys, PD_round(dp.hum_int,dp.hum_dec), rg.hum_val, rg.hum_hys);
393
394 // Send updated data via UART
395 transmit_uart_data();
396 }
397 }
398
399 // It will never happen.
400 return 0;
401}
