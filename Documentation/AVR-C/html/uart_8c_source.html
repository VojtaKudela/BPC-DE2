9* DESCRIPTION:
10* An interrupt is generated when the UART has finished transmitting or
11* receiving a byte. The interrupt handling routines use circular buffers
12* for buffering received and transmitted data.

16* power of 2.

360ISR(UART0_RECEIVE_INTERRUPT)
361
362/*************************************************************************
363 * Function: UART Receive Complete interrupt
364 * Purpose: called when the UART has received a character
365 **************************************************************************/
366{
367 unsigned char tmphead;
368 unsigned char data;
369 unsigned char usr;
370 unsigned char lastRxError = 0;
371
372
373 /* read UART status register and UART data register */
374 usr = UART0_STATUS;
375 data = UART0_DATA;
376
377 /* get FEn (Frame Error) DORn (Data OverRun) UPEn (USART Parity Error) bits */
378 #if defined(FE) && defined(DOR) && defined(UPE)
379 lastRxError = usr & (_BV(FE) | _BV(DOR) | _BV(UPE) );
380 #elif defined(FE0) && defined(DOR0) && defined(UPE0)
381 lastRxError = usr & (_BV(FE0) | _BV(DOR0) | _BV(UPE0) );
382 #elif defined(FE1) && defined(DOR1) && defined(UPE1)
383 lastRxError = usr & (_BV(FE1) | _BV(DOR1) | _BV(UPE1) );
384 #elif defined(FE) && defined(DOR)
385 lastRxError = usr & (_BV(FE) | _BV(DOR) );
386 #endif
387
388 /* calculate buffer index */
389 tmphead = ( UART_RxHead + 1) & UART_RX_BUFFER_MASK;
390
391 if (tmphead == UART_RxTail)
392 {
393 /* error: receive buffer overflow */
394 lastRxError = UART_BUFFER_OVERFLOW >> 8;
395 }
396 else
397 {
398 /* store new index */
399 UART_RxHead = tmphead;
400 /* store received data in buffer */
401 UART_RxBuf[tmphead] = data;
402 }
403 UART_LastRxError |= lastRxError;
404}

407ISR(UART0_TRANSMIT_INTERRUPT)
408
409/*************************************************************************
410 * Function: UART Data Register Empty interrupt
411 * Purpose: called when the UART is ready to transmit the next byte
412 **************************************************************************/
413{
414 unsigned char tmptail;
415
416
417 if (UART_TxHead != UART_TxTail)
418 {
419 /* calculate and store new buffer index */
420 tmptail = (UART_TxTail + 1) & UART_TX_BUFFER_MASK;
421 UART_TxTail = tmptail;
422 /* get one byte from buffer and write it to UART */
423 UART0_DATA = UART_TxBuf[tmptail]; /* start transmission */
424 }
425 else
426 {
427 /* tx buffer empty, disable UDRE interrupt */
428 UART0_CONTROL &= ~_BV(UART0_UDRIE);
429 }
430}

433/*************************************************************************
434 * Function: uart_init()
435 * Purpose: initialize UART and set baudrate
436 * Input: baudrate using macro UART_BAUD_SELECT()
437 * Returns: none
438 **************************************************************************/
439void uart_init(unsigned int baudrate)
440{
441 UART_TxHead = 0;
442 UART_TxTail = 0;
443 UART_RxHead = 0;
444 UART_RxTail = 0;
445
446 #ifdef UART_TEST
447 # ifndef UART0_BIT_U2X
448 # warning "UART0_BIT_U2X not defined"
449 # endif
450 # ifndef UART0_UBRRH
451 # warning "UART0_UBRRH not defined"
452 # endif
453 # ifndef UART0_CONTROLC
454 # warning "UART0_CONTROLC not defined"
455 # endif
456 # if defined(URSEL) || defined(URSEL0)
457 # ifndef UART0_BIT_URSEL
458 # warning "UART0_BIT_URSEL not defined"
459 # endif
460 # endif
461 #endif /* ifdef UART_TEST */
462
463 /* Set baud rate */
464 if (baudrate & 0x8000)
465 {
466 #if UART0_BIT_U2X
467 UART0_STATUS = (1 << UART0_BIT_U2X); // Enable 2x speed
468 #endif
469 }
470 #if defined(UART0_UBRRH)
471 UART0_UBRRH = (unsigned char) ((baudrate >> 8) & 0x80);
472 #endif
473 UART0_UBRRL = (unsigned char) (baudrate & 0x00FF);
474
475 /* Enable USART receiver and transmitter and receive complete interrupt */
476 UART0_CONTROL = _BV(UART0_BIT_RXCIE) | (1 << UART0_BIT_RXEN) | (1 << UART0_BIT_TXEN);
477
478 /* Set frame format: asynchronous, 8data, no parity, 1stop bit */
479 #ifdef UART0_CONTROLC
480 # ifdef UART0_BIT_URSEL
481 UART0_CONTROLC = (1 << UART0_BIT_URSEL) | (1 << UART0_BIT_UCSZ1) | (1 << UART0_BIT_UCSZ0);
482 # else
483 UART0_CONTROLC = (1 << UART0_BIT_UCSZ1) | (1 << UART0_BIT_UCSZ0);
484 # endif
485 #endif
486}/* uart_init */

488/*************************************************************************
489 * Function: uart_getc()
490 * Purpose: return byte from ringbuffer
491 * Returns: lower byte: received byte from ringbuffer
492 * higher byte: last receive error
493 **************************************************************************/
494unsigned int uart_getc(void)
495{
496 unsigned char tmptail;
497 unsigned char data;
498 unsigned char lastRxError;
499
500
501 if (UART_RxHead == UART_RxTail)
502 {
503 return UART_NO_DATA; /* no data available */
504 }
505
506 /* calculate buffer index */
507 tmptail = (UART_RxTail + 1) & UART_RX_BUFFER_MASK;
508
509 /* get data from receive buffer */
510 data = UART_RxBuf[tmptail];
511 lastRxError = UART_LastRxError;
512
513 /* store buffer index */
