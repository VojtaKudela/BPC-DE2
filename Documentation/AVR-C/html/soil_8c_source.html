14 * Function: SOIL_init()
15 * Purpose: Initialize valve pin, turn valve off.
16 * Input(s): reg - Address of Port Register, such as &PORTB.
17 * pin - Pin designation in the interval 0 to 7.
18 * Returns: none
19 ***********************************************************/
20void SOIL_init(volatile uint8_t *reg, uint8_t pin)
21{
22 // set pin to output
23 *reg |= (1<<pin);
24 // set low level on valve pin

66{

69 if ((soil_sw == 0) & (current_soil < (setpoint_soil - hyst_soil)))
